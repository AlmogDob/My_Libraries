643 double mat2D_make_identity(Mat2D m)
644 {
645  /* make identity matrix using Gauss elimination */
646  /* preforming Gauss elimination: https://en.wikipedia.org/wiki/Gaussian_elimination */
647  /* returns the factor multiplying the determinant */
648 
649  double factor_to_return = 1;
650 
651  for (size_t i = 0; i < (size_t)fmin(m.rows-1, m.cols); i++) {
652  /* check if it is the biggest first number (absolute value) */
653  size_t biggest_r = i;
654  for (size_t index = i; index < m.rows; index++) {
655  if (fabs(MAT2D_AT(m, index, index)) > fabs(MAT2D_AT(m, biggest_r, 0))) {
656  biggest_r = index;
657  }
658  }
659  if (i != biggest_r) {
660  mat2D_swap_rows(m, i, biggest_r);
661  factor_to_return *= -1;
662  }
663  for (size_t j = i+1; j < m.cols; j++) {
664  double factor = 1 / MAT2D_AT(m, i, i);
665  mat2D_sub_row_time_factor_to_row(m, j, i, MAT2D_AT(m, j, i) * factor);
666  mat2D_mult_row(m, i, factor);
667  factor_to_return *= factor;
668  }
669  }
670  double factor = 1 / MAT2D_AT(m, m.rows-1, m.cols-1);
671  mat2D_mult_row(m, m.rows-1, factor);
672  factor_to_return *= factor;
673  for (size_t c = m.cols-1; c > 0; c--) {
674  for (int r = c-1; r >= 0; r--) {
675  double factor = 1 / MAT2D_AT(m, c, c);
676  mat2D_sub_row_time_factor_to_row(m, r, c, MAT2D_AT(m, r, c) * factor);
677  }
678  }
679 
680 
681  return factor_to_return;
682 }

1013 double mat2D_triangulate(Mat2D m)
1014 {

1016  /* returns the factor multiplying the determinant */
1017

1019 
1020  for (size_t i = 0; i < (size_t)fmin(m.rows-1, m.cols); i++) {
1021  if (!MAT2D_AT(m, i, i)) { /* swapping only if it is zero */
1022  /* finding biggest first number (absolute value) */
1023  size_t biggest_r = i;
1024  for (size_t index = i; index < m.rows; index++) {
1025  if (fabs(MAT2D_AT(m, index, i)) > fabs(MAT2D_AT(m, biggest_r, i))) {
1026  biggest_r = index;
1027  }
1028  }
1029  if (i != biggest_r) {
1030  mat2D_swap_rows(m, i, biggest_r);
1031  }
1032  }

1034  double factor = 1 / MAT2D_AT(m, i, i);
1035  if (!isfinite(factor)) {

1039  mat2D_sub_row_time_factor_to_row(m, j, i, mat_value * factor);
